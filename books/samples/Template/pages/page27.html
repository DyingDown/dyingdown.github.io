FACTORIAL OF 2

int getFactorialofTwo(int n) {  // n > 0
    return 1 << n;  // 2 的 n 次方
}

IS FACTORIAL OF 2

bool isFactorialofTwo(int n) {
    return n > 0 ? (n & (n - 1)) == 0 : false;
    // return n > 0 && (n & (n - 1)) == 0; simple way
    /* 如果是 2 的幂，n 一定是 100... n-1 就是 1111.... 所以做与运算结果为 0 */
}

AVERAGE OF TOW NUMBERS

int getAverage(int x, int y) {
    return (x + y) >> 1;
