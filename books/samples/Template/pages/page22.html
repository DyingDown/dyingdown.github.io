CHAIN FORWARD STAR 

#include<iostream>
#include<string.h>
#include<limits>
    
const int maxn = 10005;
const int maxm = 1000005;//edge
using namespace std;
int n;
struct node {
    int to, next;
};
node edge[maxm];
int box[maxn]; //box[i]  节点i下第一条边
int ecnt;//边的个数
void _make_map(int from, int to){
    edge[ecnt].to = to; // to 节点
    // 同节点下该边下一条边
    edge[ecnt].next = box[from];
    // 节点from的第一条边
    box[from] = ecnt ++; 
}
