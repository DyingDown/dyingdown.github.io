bool solve(int v, int L, int R, int *a){   
    int mid; 
    mid = (R - L) / 2 + L; 
    while(L <= R){
        if(a[mid] == v) 
            return true; 
        else if(a[mid] < v) 
            L = mid; 
        else 
            R = mid; 
    } 
    return false; 
}
