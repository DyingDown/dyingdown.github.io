int main(){
    int m, n, t;
    long long sum;
    cin >> t;
    while(t --){
        cin>>n>>m;
            sum=1;
        for(int i = 1; i <= m; i ++){
            sum *= n --;
            sum /= i;
        }    
        printf("%lld\n",sum);
    }
    return 0;
}
